CREATE TABLE CLIENTE 
( 
 cod_cliente INT PRIMARY KEY AUTO_INCREMENT,  
 nome CHAR(n) NOT NULL,  
 endereço CHAR(n) NOT NULL,  
 cep CHAR(n) NOT NULL,  
 bairro CHAR(n) NOT NULL,  
 cidade CHAR(n) NOT NULL,  
 estado CHAR(n) NOT NULL,  
 telefone VARCHAR(n) NOT NULL,  
 email CHAR(n) NOT NULL,  
 idPESSOA FÍSICA INT,  
 idPESSOA JURÍDICA INT,  
); 

CREATE TABLE PEDIDO 
( 
 idCLIENTE INT,  
 numero_pedido INT PRIMARY KEY AUTO_INCREMENT,  
 data_pedido DATE NOT NULL,  
 hora_pedido CHAR(n) NOT NULL,  
 idFUNCIONÁRIOS INT,  
); 

CREATE TABLE FUNCIONÁRIOS 
( 
 cod_funcionario INT PRIMARY KEY AUTO_INCREMENT,  
 cod_setor INT NOT NULL,  
 nome CHAR(n) NOT NULL,  
 endereço CHAR(n) NOT NULL,  
 cep INT NOT NULL,  
 bairro CHAR(n) NOT NULL,  
 cidade CHAR(n) NOT NULL,  
 estado VARCHAR(n) NOT NULL,  
 telefone VARCHAR(n) NOT NULL,  
 email CHAR(n) NOT NULL,  
 data_admissao DATE NOT NULL,  
 Setor CHAR(n),  
 CHECK (Setor LIKE '%%')
); 

CREATE TABLE PRODUTOS 
( 
 cod_produto INT PRIMARY KEY AUTO_INCREMENT,  
 quantidade INT NOT NULL,  
 preço_unit FLOAT NOT NULL,  
 nome CHAR(n) NOT NULL,  
 descrição VARCHAR(n) NOT NULL,  
 tipo VARCHAR(n) NOT NULL,  
 idFUNCIONÁRIOS INT,  
); 

CREATE TABLE LISTA DE ITENS 
( 
 idPEDIDO INT,  
 idPRODUTOS INT,  
); 

CREATE TABLE PESSOA FÍSICA 
( 
 cpf VARCHAR(n) NOT NULL,  
 rg VARCHAR(n) NOT NULL,  
); 

CREATE TABLE PESSOA JURÍDICA 
( 
 cnpj VARCHAR(n) NOT NULL,  
 responsável CHAR(n) NOT NULL,  
); 

ALTER TABLE CLIENTE ADD FOREIGN KEY(idPESSOA FÍSICA) REFERENCES PESSOA FÍSICA (idPESSOA FÍSICA)
ALTER TABLE CLIENTE ADD FOREIGN KEY(idPESSOA JURÍDICA) REFERENCES PESSOA JURÍDICA (idPESSOA JURÍDICA)
ALTER TABLE PEDIDO ADD FOREIGN KEY(idCLIENTE) REFERENCES CLIENTE (idCLIENTE)
ALTER TABLE PEDIDO ADD FOREIGN KEY(idFUNCIONÁRIOS) REFERENCES FUNCIONÁRIOS (idFUNCIONÁRIOS)
ALTER TABLE PRODUTOS ADD FOREIGN KEY(idFUNCIONÁRIOS) REFERENCES FUNCIONÁRIOS (idFUNCIONÁRIOS)
ALTER TABLE LISTA DE ITENS ADD FOREIGN KEY(idPEDIDO) REFERENCES PEDIDO (idPEDIDO)
ALTER TABLE LISTA DE ITENS ADD FOREIGN KEY(idPRODUTOS) REFERENCES PRODUTOS (idPRODUTOS)
